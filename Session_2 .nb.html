library(tidyverse)

Beziehung zwischen Anzahl der Bcher/Verlag und deren Bewertungen

###Neue Tabelle mit ISBN und Verlag

ISBNPublisher <- BX_Books %>%
  select(ISBN, Publisher)

###Anzahl der Bcher pro Verlag

AnzahlBooks <- BX_Books %>%
  select(Publisher, ISBN) %>%
  group_by(Publisher) %>%
  summarise(AnzahlDerBcher = n())
`summarise()` ungrouping output (override with `.groups` argument)

###Zusammenfhren von Bcher und deren Bewertungen

gemeinsam <- merge(BX_Book_Ratings, BX_Books)

###Die Bewertungen der Bcher, sortiert nach Verlag

AnzahlRatings <- merge(BX_Book_Ratings, BX_Books) %>%
  group_by(Publisher) %>%
  summarise(AnzahlDerRatings = n())
`summarise()` ungrouping output (override with `.groups` argument)

###Relation der Anzahl von Bchern und Bewertungen

Relation <- merge(AnzahlBooks, AnzahlRatings) %>%
  select(AnzahlDerBcher, AnzahlDerRatings)
plot(Relation)

Zeitalter der Bcher und der Einfluss auf deren Bewertung

###Verffentlichung und ISBN

AlterBooks <- BX_Books %>%
  select(`Year-Of-Publication`, ISBN) %>%
  group_by(`Year-Of-Publication`) %>%
  summarise(AlterDerBcher = n())
`summarise()` ungrouping output (override with `.groups` argument)

###Einflieen der Bewertung der Bcher

AlterBooks <- merge(BX_Books, BX_Book_Ratings) %>%
  group_by(`Year-Of-Publication`) %>%
  summarise(BucherInDemJahr = n()) %>%
  filter(`Year-Of-Publication` > 1500)
`summarise()` ungrouping output (override with `.groups` argument)
plot(AlterBooks)
